Code Examples

Variables & Functions

// Variables with dynamic typing
let name = "Flux"
let version = 1.0
let isActive = true

// First-class functions
fun greet(name) {
    return "Hello, " + name + "!"
}

print greet("World")  // Hello, World!

Control Flow

// Conditional statements
if (version >= 1.0) {
    print "Stable release!"
} else {
    print "Beta version"
}

// Loops and recursion
fun factorial(n) {
    if (n <= 1) return 1
    return n * factorial(n - 1)
}

print factorial(5)  // 120

Closures & Advanced Features

// Closures and lexical scoping
fun makeCounter() {
    let count = 0
    fun increment() {
        count = count + 1
        return count
    }
    return increment
}

let counter = makeCounter()
print counter()  // 1
print counter()  // 2

// Built-in functions
print sqrt(16)   // 4
print abs(-42)   // 42

Getting Started

1. Download & Build

Download the source code and build with the included build scripts.

./build.bat # Windows
make all # Unix/Linux

2. Interactive Mode

Start the REPL to experiment with Flux interactively.

./flux.exe # Start REPL

3. Run Scripts

Execute Flux programs from files with example programs included.

./flux.exe examples/hello.flux
